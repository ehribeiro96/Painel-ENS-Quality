===== /tmp/ens-unificado-analysis/projeto-ens-unificado-main/services/artifact-server/Dockerfile =====
FROM node:20-alpine

WORKDIR /app

COPY package.json ./
RUN apk add --no-cache su-exec

COPY src ./src
COPY docker-entrypoint.sh /usr/local/bin/artifact-server-entrypoint.sh

RUN mkdir -p /app/data && \
    chown -R node:node /app && \
    chmod +x /usr/local/bin/artifact-server-entrypoint.sh

EXPOSE 8095

ENTRYPOINT ["/usr/local/bin/artifact-server-entrypoint.sh"]
CMD ["node", "src/server.js"]

===== /tmp/ens-unificado-analysis/projeto-ens-unificado-main/services/artifact-server/docker-entrypoint.sh =====
#!/usr/bin/env sh
set -eu

mkdir -p /app/data/metadata /app/data/objects /app/data/tmp
chown -R node:node /app/data

exec su-exec node "$@"
